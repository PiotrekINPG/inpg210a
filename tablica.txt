#include <stdio.h>
#include <stdlib.h>

void deleteMatrix(int*** ptab, int n)
{
  int i;
  for(i=0;i<n;i++)
        free((*ptab)[i]);
  free(*ptab);
  *ptab=NULL;
}

int **createMatrix(int n)
{
    int**tab=malloc(n*sizeof(*tab));
    if(tab==NULL) return NULL;

    int i;
    for(i=0;i<n;i++)
    {
        tab[i]=malloc(n*sizeof(**tab));
        if(tab==NULL)
        {
            deleteMatrix(&tab,i);
            return NULL;
        }
    }
    return tab;
}

void copyMatrix(int*** pdest, int** src, int n)
{
    int **dest=createMatrix(n);
    if(dest==NULL)
        return;
    int i,j;
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            dest[i][j]=src[i][j];
        }
    }
    *pdest=dest;
}

void fillMatrix( int**tab,int n)
    {
     int i,j,k=1;
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {
               tab[i][j]=k;
               k++;
            }
        }

    }
